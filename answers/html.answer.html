                background-color: lightblue;
            }

                color: darkblue;
            }
        </style>
    </head>
    <body>
        <h1>
            Welcome to My Webpage
        </h1>

        <p id="intro">
            Hello, this is an example paragraph.
        </p>

        <button onclick="changeText()">
            Click Me
        </button>

        <script>
            // JavaScript code goes here
            function changeText() {
                var para = document.getElementById("intro");

            }
        </script>
