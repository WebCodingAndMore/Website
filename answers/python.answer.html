# Classes and Objects
# Learning Objective: Learn the basics of object-oriented programming in Python.

# Define a class named 'Animal'
class Animal:
    # Initialize the class with 'name' and 'sound' attributes
    def __init__(self, name, sound):
        self.name = name      # The name of the animal (string)
        self.sound = sound    # The sound the animal makes (string)

    # Define a method to make the animal sound
    def make_sound(self):
        print(f"The {self.name} says {self.sound}")

# Create an instance of 'Animal' representing a dog
dog = Animal("dog", "woof")  # e.g., "dog", "woof"

# Call the 'make_sound' method
dog.make_sound()
Expected Output: 
The dog says woof
