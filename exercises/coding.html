Basic Programming Concepts Across Multiple Languages

Concept 1: Variables and Printing

In this section, you'll see how different programming languages assign values to variables and print output. Pay attention to how each language uses different syntax but follows the same fundamental concept.

Python

# Python uses a simple syntax to assign variables
name = "Alice"
age = 25
# Print the values using the print() function
print("Name:", name)
print("Age:", age)

JavaScript

// JavaScript assigns variables using 'let' or 'const'
let name = "Alice";
let age = 25;
// Output values using console.log
console.log("Name:", name);
console.log("Age:", age);

C++

// C++ uses 'string' and 'int' for variable types
#include <iostream>
using namespace std;

int main() {
    string name = "Alice";
    int age = 25;
    // Use cout to print the values
    cout << "Name: " << name << endl;
    cout << "Age: " << age << endl;
    return 0;
}

Rust

// Rust uses let to declare variables
fn main() {
    let name = "Alice";
    let age = 25;
    // Print using println!
    println!("Name: {}", name);
    println!("Age: {}", age);
}

C#

// C# uses string and int
using System;

class Program {
    static void Main() {
        string name = "Alice";
        int age = 25;
        Console.WriteLine("Name: " + name);
        Console.WriteLine("Age: " + age);
    }
}

Go

// Go uses var or := for variable declaration
package main
import "fmt"

func main() {
    name := "Alice"
    age := 25
    // Print the values using fmt.Println
    fmt.Println("Name:", name)
    fmt.Println("Age:", age)
}

Perl

# Perl declares variables using my
my $name = "Alice";
my $age = 25;
# Print values using print
print "Name: $name\n";
print "Age: $age\n";

BASIC

REM BASIC uses simple PRINT statements
name = "Alice"
age = 25
REM Print values using PRINT
PRINT "Name: "; name
PRINT "Age: "; age

Ruby

# Ruby uses simple variable assignment
name = "Alice"
age = 25
# Print values using puts
puts "Name: #{name}"
puts "Age: #{age}"
Expected Output:
Name: Alice
Age: _____  (Hint: Fill in the number!)

Concept 2: Conditionals

In this section, you’ll see how different programming languages handle conditional statements using "if-else" to make decisions.

Python

  num = 10
  if num % 2 == 0:
      print(f"{num} is even")
  else:
      print(f"{num} is odd")

JavaScript

  let num = 10;
  if (num % 2 === 0) {
      console.log(num + " is even");
  } else {
      console.log(num + " is odd");
  }

C++

  int num = 10;
  if (num % 2 == 0) {
      cout << num << " is even" << endl;
  } else {
      cout << num << " is odd" << endl;
  }

C#

  int num = 10;
  if (num % 2 == 0) {
      Console.WriteLine(num + " is even");
  } else {
      Console.WriteLine(num + " is odd");
  }

Rust

  fn main() {
      let num = 10;
      if num % 2 == 0 {
          println!("{} is even", num);
      } else {
          println!("{} is odd", num);
      }
  }

Go

  package main
  import "fmt"
  
  func main() {
      num := 10
      if num % 2 == 0 {
          fmt.Println(num, "is even")
      } else {
          fmt.Println(num, "is odd")
      }
  }

Perl

  my $num = 10;
  if ($num % 2 == 0) {
      print "$num is even\n";
  } else {
      print "$num is odd\n";
  }

BASIC

  REM BASIC conditionals use IF THEN ELSE
  num = 10
  REM check if number is even or odd
  IF num MOD 2 = 0 THEN
      PRINT num & " is even"
  ELSE
      PRINT num & " is odd"
  END IF

Ruby

  num = 10
  if num % 2 == 0
      puts "#{num} is even"
  else
      puts "#{num} is odd"
  end
Expected Output:
  10 is _____  (Hint: Fill in "even" or "odd")

Concept 3: Loops

In this section, you'll learn how different programming languages handle loops, allowing you to repeat tasks multiple times.

Python

  for i in range(1, 4):
      print(f"Loop iteration: {i}")

JavaScript

  for (let i = 1; i <= 3; i++) {
      console.log("Loop iteration: " + i);
  }

C++

  for (int i = 1; i <= 3; i++) {
      cout << "Loop iteration: " << i << endl;
  }

C#

  for (int i = 1; i <= 3; i++) {
      Console.WriteLine("Loop iteration: " + i);
  }

Rust

  fn main() {
      for i in 1..=3 {
          println!("Loop iteration: {}", i);
      }
  }

Go

  package main
  import "fmt"
  
  func main() {
      for i := 1; i <= 3; i++ {
          fmt.Println("Loop iteration: ", i)
      }
  }

Perl

  for $i (1..3) {
      print "Loop iteration: $i\n";
  }

BASIC

  FOR i = 1 TO 3
      PRINT "Loop iteration: "; i
  NEXT i

Ruby

  1.upto(3) do i
      puts "Loop iteration: #{i}"
  end
Expected Output:
  Loop iteration: 1
  Loop iteration: 2
  Loop iteration: 3

Concept 4: Functions

In this section, you’ll learn how different programming languages define and call functions to perform tasks.

Python

  def greet(name):
      print(f"Hello, {name}!")
  
  greet("Alice")

JavaScript

  function greet(name) {
      console.log("Hello, " + name + "!");
  }
  
  greet("Alice");

C++

  void greet(string name) {
      cout << "Hello, " << name << "!" << endl;
  }
  
  int main() {
      greet("Alice");
      return 0;
  }

C#

  void Greet(string name) {
      Console.WriteLine("Hello, " + name + "!");
  }
  
  Greet("Alice");

Rust

  fn greet(name: &str) {
      println!("Hello, {}!", name);
  }
  
  fn main() {
      greet("Alice");
  }

Go

  package main
  import "fmt"
  
  func greet(name string) {
      fmt.Println("Hello, ", name, "!")
  }
  
  greet("Alice")

Perl

  sub greet {
      my $name = $_[0];
      print "Hello, $name!\n";
  }
  
  greet("Alice");

BASIC

  REM BASIC subroutine for greeting
  SUB GREET(name)
      PRINT "Hello, "; name; "!"
  END SUB
  
  CALL GREET("Alice")

Ruby

  def greet(name)
      puts "Hello, #{name}!"
  end
  
  greet("Alice")
Expected Output:
  Hello, Alice!

Concept 5: Data Types

In this section, you’ll learn how different programming languages handle basic data types like strings, integers, floats, and booleans.

Python

name = "Alice"   # string
age = 25         # integer
height = 1.75    # float
is_student = True # boolean

JavaScript

let name = "Alice";  // string
let age = 25;        // integer
let height = 1.75;   // float
let isStudent = true; // boolean

C++

string name = "Alice";  // string
int age = 25;          // integer
float height = 1.75;    // float
bool is_student = true; // boolean

C#

string name = "Alice";  // string
int age = 25;           // integer
double height = 1.75;   // float
bool isStudent = true; // boolean

Rust

let name = "Alice";   // string
let age = 25;         // integer
let height = 1.75;    // float
let is_student = true; // boolean

Go

var name = "Alice"  // string
var age = 25         // integer
var height = 1.75    // float
var isStudent = true  // boolean

Perl

my $name = "Alice";   # string
my $age = 25;         # integer
my $height = 1.75;    # float
my $is_student = 1;  # boolean (1 for true)

BASIC

REM BASIC types are simple assignments
name = "Alice"
age = 25
height = 1.75
isStudent = 1 REM 1 for true

Ruby

name = "Alice"    # string
age = 25          # integer
height = 1.75     # float
is_student = true  # boolean
Expected Output:
Name: Alice
Age: 25
Height: 1.75
Is student: true

Concept 6: Arrays (or Lists)

In this section, you’ll see how different programming languages handle arrays (also called lists in some languages).

Python

  fruits = ["apple", "banana", "cherry"]
  print(fruits)

JavaScript

  let fruits = ["apple", "banana", "cherry"];
  console.log(fruits);

C++

  string fruits[] = {"apple", "banana", "cherry"};
  cout << fruits[0] << ", " << fruits[1] << ", " << fruits[2];

C#

  string[] fruits = {"apple", "banana", "cherry"};
  Console.WriteLine(string.Join(", ", fruits));

Rust

  let fruits = ["apple", "banana", "cherry"];
  println!("{:?}", fruits);

Go

  var fruits = []string{"apple", "banana", "cherry"}
  fmt.Println(fruits)

Perl

  my @fruits = ("apple", "banana", "cherry");
  print join(", ", @fruits);

BASIC

  REM BASIC does not have advanced arrays, so we use strings
  fruits = "apple, banana, cherry"
  PRINT fruits

Ruby

  fruits = ["apple", "banana", "cherry"]
  puts fruits.join(", ")
Expected Output:
  apple, banana, cherry

Concept 7: User Input

In this section, you’ll learn how different programming languages handle user input.

Python

  name = input("Enter your name: ")
  print("Hello, " + name)

JavaScript

  let name = prompt("Enter your name:");
  alert("Hello, " + name);

C++

  string name;
  cout << "Enter your name: ";
  cin >> name;
  cout << "Hello, " << name << "!";

C#

  Console.Write("Enter your name: ");
  string name = Console.ReadLine();
  Console.WriteLine("Hello, " + name);

Rust

  use std::io;
  
  fn main() {
      let mut name = String::new();
      println!("Enter your name: ");
      io::stdin().read_line(&mut name).unwrap();
      println!("Hello, {}!", name);
  }

Go

  var name string
  fmt.Print("Enter your name: ")
  fmt.Scanln(&name)
  fmt.Println("Hello, ", name)

Perl

  print "Enter your name: ";
  my $name = <>;
  chomp($name);
  print "Hello, $name\n";

BASIC

  REM BASIC gets input using INPUT command
  INPUT "Enter your name: ", name
  PRINT "Hello, "; name;

Ruby

  print "Enter your name: "
  name = gets.chomp
  puts "Hello, #{name}"
Expected Output:
  Enter your name: _____
  Hello, _____!

Concept 8: String Manipulation

In this section, you’ll see how different programming languages handle string manipulation tasks like concatenation and formatting.

Python

  name = "Alice"
  age = 25
  sentence = f"My name is {name} and I am {age} years old."
  print(sentence)

JavaScript

  let name = "Alice";
  let age = 25;
  let sentence = `My name is ${name} and I am ${age} years old.`;
  console.log(sentence);

C++

  string name = "Alice";
  int age = 25;
  cout << "My name is " << name << " and I am " << age << " years old.";

C#

  string name = "Alice";
  int age = 25;
  Console.WriteLine("My name is {0} and I am {1} years old.", name, age);

Rust

  let name = "Alice";
  let age = 25;
  println!("My name is {} and I am {} years old.", name, age);

Go

  var name = "Alice"
  var age = 25
  fmt.Printf("My name is %s and I am %d years old.\n", name, age)

Perl

  my $name = "Alice";
  my $age = 25;
  print "My name is $name and I am $age years old.\n";

BASIC

  PRINT "My name is "; name; " and I am "; age; " years old."

Ruby

  name = "Alice"
  age = 25
  puts "My name is #{name} and I am #{age} years old."
Expected Output:
  My name is Alice and I am 25 years old.
